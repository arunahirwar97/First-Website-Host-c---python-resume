It is simple calculater using c++ programming. So it is four option and choose any option when addition ,multipication,division and subtraction and after the enter value of a and b so find the result . 


    #include< iostream.h>

    void main()
    {

         cout<<"Enter an operater(+,-,*,/) : ";
         cin>>a;
         cout<<"\nEnter two number :";
         cin>>b>>c;
         cout<<"\n\n\n";
         switch(a)
            {
                case '+': cout<< b<<"+"<< c << "="<< b+c;
	            break;
                case '-': cout<< b<<"-"<< c << "="<< b-c;

                case '*': cout<< b<<"*"<< c<<"="<< b*c;

                case '/': cout<< b<<"/"<< c<<"="<< b/c;
