       for(i=0;i<=n-1;i++)

           min=a[i];
           loc=i;
           for(j=i+1;j<=n-1;j++)
             {
               if(a[j]< min)
                 {
                   min=a[j];
                   loc=j;
                  }
              }
           if(loc!=1)
            {
              temp=a[i];
              a[i]=a[loc];
              a[loc]=temp;
            }
         }
       cout<<"The number after selection sorting are :"<< endl;
       for(i=0;i< =n-1;i++)

           cout<<"\n"<< a[i];
